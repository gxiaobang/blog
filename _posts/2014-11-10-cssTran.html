用js控制css3 Transition做各种高难等动作。 

	
	/**
	 * css3Tran(options);
	 * options配置属性
	 * 	elem		目标节点
	 * 	from		起始状态
	 * 	to  		目标状态
	 * 	duration	持续时间
	 * 	func		动画缓动函数
	 */
	css3Tran({
			elem: '#box',
			to: {
				width: 200
			}
		})
		.then({
			to: {
				transform: 'scale(2)',
				opacity: 0
			}
		})
		.complete(function() {
			new Dialog({log: '动画完成'});
		});
