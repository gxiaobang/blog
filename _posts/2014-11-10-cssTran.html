transition可以完成各种炫酷的动画效果。如果单纯的只是想做个鼠标悬浮，css就可以完成，但是如果想完成复杂点的功能，需要javascript来控制css的属性。 

	
	/**
	 * css3Tran(options);
	 * options配置属性
	 * 	elem		目标节点
	 * 	from		起始状态
	 * 	to  		目标状态
	 * 	duration	持续时间
	 * 	func		运动缓动函数
	 */
	css3Tran({
			elem: '#box',
			to: {
				width: 200
			}
		})
		.then({
			to: {
				transform: 'scale(2)',
				opacity: 0
			}
		})
		.complete(function() {
			new Dialog({log: '动画完成'});
		});
